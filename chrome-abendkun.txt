{
  "name": "JS Line Throw Injector",
  "version": "1.0",
  "manifest_version": 3,
  "permissions": [
    "debugger",
    "tabs"
  ],
  "host_permissions": [
    "https://example.com/*"
  ],
  "background": {
    "service_worker": "background.js"
  },
  "action": {}
}





const TARGET_PAGE_URL_PREFIX = "https://example.com/target";

// scriptParsed に出ている JS ファイル名に合わせる
// 例: https://example.com/assets/app.abc123.js
const TARGET_SCRIPT_REGEX = /app\..*\.js$/;

// この行を実行せず、直前で throw する
// DevTools で見えている JS 側の行番号
const TARGET_LINE_1_BASED = 88;

const ERROR_MESSAGE = "Injected error before target line";

const attachedTabs = new Set();
const patchedScripts = new Set();



chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  console.log("[onUpdated]", {
    tabId,
    status: changeInfo.status,
    url: tab.url
  });

  if (changeInfo.status !== "complete") {
    return;
  }

  if (!tab.url || !tab.url.startsWith(TARGET_PAGE_URL_PREFIX)) {
    return;
  }

  attachDebugger(tabId);
});



chrome.tabs.onRemoved.addListener((tabId) => {
  attachedTabs.delete(tabId);

  for (const key of [...patchedScripts]) {
    if (key.startsWith(`${tabId}:`)) {
      patchedScripts.delete(key);
    }
  }
});



chrome.debugger.onDetach.addListener((source) => {
  if (!source.tabId) {
    return;
  }

  attachedTabs.delete(source.tabId);

  for (const key of [...patchedScripts]) {
    if (key.startsWith(`${source.tabId}:`)) {
      patchedScripts.delete(key);
    }
  }
});



chrome.debugger.onEvent.addListener(async (source, method, params) => {
  if (method !== "Debugger.scriptParsed") {
    return;
  }

  await onScriptParsed(source, params);
});



async function attachDebugger(tabId) {
  const target = { tabId };

  try {
    if (!attachedTabs.has(tabId)) {
      await chrome.debugger.attach(target, "1.3");
      attachedTabs.add(tabId);
      console.log("[attach] OK", tabId);
    }

    await chrome.debugger.sendCommand(target, "Debugger.enable");
    console.log("[Debugger.enable] OK", tabId);
  } catch (e) {
    console.error("[attachDebugger] failed", e);
  }
}



async function onScriptParsed(source, params) {
  const tabId = source.tabId;

  if (!tabId) {
    return;
  }

  const scriptUrl = params.url || "";

  console.log("[scriptParsed]", scriptUrl);

  if (!TARGET_SCRIPT_REGEX.test(scriptUrl)) {
    return;
  }

  const patchKey = `${tabId}:${params.scriptId}`;

  if (patchedScripts.has(patchKey)) {
    return;
  }

  patchedScripts.add(patchKey);

  console.log("[target script found]", {
    tabId,
    scriptId: params.scriptId,
    url: scriptUrl
  });

  await patchScriptSource(source, params.scriptId, scriptUrl);
}



async function patchScriptSource(source, scriptId, scriptUrl) {
  try {
    const result = await chrome.debugger.sendCommand(source, "Debugger.getScriptSource", {
      scriptId
    });

    const originalSource = result.scriptSource;

    if (!originalSource) {
      console.warn("[patch] script source is empty", scriptUrl);
      return;
    }

    const lines = originalSource.split("\n");

    const insertIndex = TARGET_LINE_1_BASED - 1;

    if (insertIndex < 0 || insertIndex >= lines.length) {
      console.warn("[patch] target line out of range", {
        targetLine: TARGET_LINE_1_BASED,
        totalLines: lines.length,
        scriptUrl
      });
      return;
    }

    const throwCode = `throw new Error(${JSON.stringify(ERROR_MESSAGE)});`;

    lines.splice(insertIndex, 0, throwCode);

    const patchedSource = lines.join("\n");

    console.log("[patch] trying setScriptSource", {
      scriptUrl,
      scriptId,
      targetLine: TARGET_LINE_1_BASED,
      insertIndex,
      beforeLine: lines[insertIndex + 1],
      throwCode
    });

    const setResult = await chrome.debugger.sendCommand(source, "Debugger.setScriptSource", {
      scriptId,
      scriptSource: patchedSource
    });

    console.log("[patch] setScriptSource result", setResult);
  } catch (e) {
    console.error("[patch] failed", {
      scriptUrl,
      scriptId,
      error: e
    });
  }
}
