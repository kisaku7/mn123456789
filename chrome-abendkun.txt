{
  "name": "TS Line Error Injector",
  "version": "1.0",
  "manifest_version": 3,
  "permissions": [
    "debugger",
    "tabs"
  ],
  "host_permissions": [
    "https://example.com/*"
  ],
  "background": {
    "service_worker": "background.js"
  },
  "action": {}
}

const TARGET_PAGE_URL_PREFIX = "https://example.com/target";

// DevTools の Sources で見えている TS ファイル名に合わせる
const TARGET_TS_FILE_REGEX = ".*TicketPage\\.ts$";

// この行を実行せずに例外で落としたい
// DevTools 表示上の行番号をそのまま書く
const TARGET_LINE_1_BASED = 88;

// CDP の lineNumber は 0 始まり
// DevTools上の88行目 = CDPでは87
const BREAKPOINT_LINE_0_BASED = TARGET_LINE_1_BASED - 1;

const ERROR_MESSAGE = "Injected error before target line";

const attachedTabs = new Set();
const breakpointSetTabs = new Set();
const thrownTabs = new Set();

chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.status !== "complete") {
    return;
  }

  if (!tab.url || !tab.url.startsWith(TARGET_PAGE_URL_PREFIX)) {
    return;
  }

  attachAndSetErrorBreakpoint(tabId);
});

chrome.tabs.onRemoved.addListener((tabId) => {
  attachedTabs.delete(tabId);
  breakpointSetTabs.delete(tabId);
  thrownTabs.delete(tabId);
});

chrome.debugger.onDetach.addListener((source) => {
  if (!source.tabId) {
    return;
  }

  attachedTabs.delete(source.tabId);
  breakpointSetTabs.delete(source.tabId);
  thrownTabs.delete(source.tabId);
});

chrome.debugger.onEvent.addListener(async (source, method, params) => {
  if (method !== "Debugger.paused") {
    return;
  }

  await onPaused(source, params);
});

async function attachAndSetErrorBreakpoint(tabId) {
  const target = { tabId };

  try {
    if (!attachedTabs.has(tabId)) {
      await chrome.debugger.attach(target, "1.3");
      attachedTabs.add(tabId);
    }

    await chrome.debugger.sendCommand(target, "Debugger.enable");

    if (!breakpointSetTabs.has(tabId)) {
      await chrome.debugger.sendCommand(target, "Debugger.setBreakpointByUrl", {
        urlRegex: TARGET_TS_FILE_REGEX,
        lineNumber: BREAKPOINT_LINE_0_BASED,
        columnNumber: 0
      });

      breakpointSetTabs.add(tabId);
    }

    console.log("[TS Line Error Injector] ready", {
      tabId,
      targetPage: TARGET_PAGE_URL_PREFIX,
      targetFile: TARGET_TS_FILE_REGEX,
      targetLine1Based: TARGET_LINE_1_BASED,
      breakpointLine0Based: BREAKPOINT_LINE_0_BASED
    });
  } catch (e) {
    console.error("[TS Line Error Injector] setup failed:", e);
  }
}

async function onPaused(source, params) {
  const tabId = source.tabId;
  const callFrame = params.callFrames && params.callFrames[0];

  if (!tabId || !callFrame) {
    return;
  }

  const location = callFrame.location;

  if (location.lineNumber !== BREAKPOINT_LINE_0_BASED) {
    await resume(source);
    return;
  }

  if (thrownTabs.has(tabId)) {
    await resume(source);
    return;
  }

  thrownTabs.add(tabId);

  try {
    await chrome.debugger.sendCommand(source, "Debugger.evaluateOnCallFrame", {
      callFrameId: callFrame.callFrameId,
      expression: `
        (() => {
          throw new Error(${JSON.stringify(ERROR_MESSAGE)});
        })()
      `
    });
  } catch (e) {
    console.error("[TS Line Error Injector] throw failed:", e);
  }
}

async function resume(source) {
  try {
    await chrome.debugger.sendCommand(source, "Debugger.resume");
  } catch (e) {
    console.error("[TS Line Error Injector] resume failed:", e);
  }
}
